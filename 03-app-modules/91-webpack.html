# تنظیمات Webpack

در ابتدا مفاهیم زیر را توضیح می دهیم Build Tool: ابزار ساخت ابزاری است جهت تولید تبدیل کد های ما به برنامه قابل اجرا - کدهای TypeScript توسط مرورگر قابل اجرا نیست Webpack : به عنوان محبوب ترین BuildTool فرانت اند میباشد و ما نیز از همین سیستم استفاده میکنیم ، webpack دارای امکانات زیاد و سطح بالایی از سفارش سازی است Bundle : فایل است که به عنوان خروجی فرآیند Build ایجاد می شوند ، خروجی Webpack ، فایل جاوااسکریپت است که قابل اجرا توسط مرورگر می باشد


 

































































const { join } = require('path');
const { statSync, writeFileSync } = require('fs');
const webpack = require('webpack');
const MinifyPlugin = require('babel-minify-webpack-plugin');
const { CheckerPlugin } = require('awesome-typescript-loader')
const makeDefaultFrontBuildConfigFile = filepath => {
    const result = require(join(__dirname, 'frontend-build-config-default'));
    writeFileSync(filepath, JSON.stringify(result), { encoding: 'utf-8' });
    return result;
}
function fileExists(filePath) {
    try { return !!(statSync(filePath).size); }
    catch{ return false; }
}
const isProd = (process.env.NODE_ENV == 'production');
const buildConfigFileName = join(__dirname, 'frontend-build-config.json');
const buildConfig = fileExists(buildConfigFileName) ? require(buildConfigFileName) : makeDefaultFrontBuildConfigFile(buildConfigFileName);
const rules = [
    {
        test: /\.tsx?$/, use: [
            {
                loader: 'ts-loader',
                options: {
                    transpileOnly: true
                }
            }
        ]
    },
    { test: /\.png$/, loader: "url-loader?mimetype=image/png" },
    { test: /\.jpg$/, loader: "url-loader?mimetype=image/jpeg" },
    { test: /\.svg$/, loader: 'svg-inline-loader' },
];
const plugins = [
    new CheckerPlugin(),
    isProd && new MinifyPlugin(),
    !isProd && new webpack.NamedModulesPlugin(),
    new webpack.DefinePlugin({ 'process.env.NODE_ENV': JSON.stringify(isProd ? 'production' : 'development') })
].filter(x => !!x);
const entry = {
    domain: './src/domain/domain.tsx',
    'domain-FA_IR': './src/domain/domain-FA_IR.tsx',
    'domain-EN_US': './src/domain/domain-EN_US.tsx',
    //   'test-cases': './src/domain/test-cases.ts'
};
const absPath = path => path.startsWith('./') ? join(process.cwd(), path) : path;
const extensions = ['.ts', '.tsx', '.jpg', '.png'];
const externals = {
    '@organic-ui': 'OrganicUI',
    '@reinvent': 'reinvent',
    '@reinvent/test': 'organicTest',  
    'react':'React',
    'react-dom':'ReactDOM'
}
const dist= absPath(buildConfig.dist || join(__dirname, 'public/bundle') );
console.log('dist>>>',dist);
module.exports = {
    entry,
    output: {
        path:dist,
        filename: '[name].js',
        publicPath: '/'
    },
    resolve: { extensions },
    externals,
    module: { rules }, plugins
};
