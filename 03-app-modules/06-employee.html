function openDialog(templateType: TemplateTypes, initialData) {
    const action = actionByTemplateType[templateType];
    if (!action) throw `invalid actionByTemplateType=> ${templateType}`;
    return action(initialData instanceof Array ? {} : initialData, initialData instanceof Array ? initialData : null);
}

رجوع به تعاریف خودمان – ما چندین فایل مهم تعریف داریم که ساختار پروژه به آن اشاره شده است . فایل تعریف پسوند شان می باشد
- ارتباط با هسته سیستم - لینک به `API` های پرسنل و داده های سفارشی - لینک به دیالوگ های مربوط سنجه های هویتی - لینک به کامپونت پلاک خودرو - لینک به آیگون فرد
تنظیمات مربوط ماژول مشخص میکند - این ماژول از نوع CRUD می باشد - مجوز ها نیز در قسمت permissions مشخص شده است
تعریف یک ماژول `crud` بوسیله `reinvent`
مشخص میکند که بر اساس کدام نوع `template` کدام متد باید اجرا شود
