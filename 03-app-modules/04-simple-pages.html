# ماژول های ساده در فرانت اند

این ماژول ساده ترین ماژول های مرتبط با فریم ورک oragnic-ui می باشد که بوسیله reinvert پیاده سازی شده اند که بسیار شبیه بهم هستند

# ماژول درب ها


 



















































/// <reference path="../../dts/globals.d.ts" />
/// <reference path="entities.d.ts" />


import { devices as deviceListView } from "./device-view";
import { DoorsController as actions } from "./api";
import { DataLookup, Field, IOptionsForCRUD, Switch } from "@organic-ui";
import { DataList, DataPanel } from "@organic-ui";
// import { crudPermissions, DOOR_DEFINE_CREATE, DEVICE_DEFINE_DELETE, DEVICE_DEFINE_UPDATE, DEVICE_DEFINE_READ } from './permissions';
import { crudPermissions, DOOR_DEFINE_CREATE } from './permissions';

const { i18n } = OrganicUI;


const options: IOptionsForCRUD =
{
    routeForSingleView: '/view/admin/door/:id',
    routeForListView: '/view/admin/doors',
    pluralName: 'doors', singularName: "door", iconCode: 'fa-map',
    permissionKeys: crudPermissions(DOOR_DEFINE_CREATE),

};
export const doors = reinvent('frontend:crud', { options, actions })
    .singleView(({ binding }) => (<>
        <DataPanel header={i18n("primary-fields")} primary className="medium-fields" >
            <Field accessor={binding.active} >
                <Switch />
            </Field>
            <Field accessor={binding.name} required />
            <Field accessor={binding.description} />
            <Field accessor={binding.entryDeviceId} >
                <DataLookup source={deviceListView} />
            </Field>
            <Field accessor={binding.exitDeviceId}  >
                <DataLookup source={deviceListView} />
            </Field>
            <Field accessor={binding.openTimeSeconds} required />
            <Field accessor={binding.relay}  />
            <Field accessor={binding.exitButtonPort}  />
            <Field accessor={binding.doorSensorPort}  />
        </DataPanel>
    </>))
    .listView(({ binding }) => <>
        <DataList>
            <Field accessor={binding.name} />
            {/* <Field accessor="code" />
            <Field accessor="serial" />
            <Field accessor="model" required>
                <ComboBox items={DeviceModels.Models} />
            </Field> */}
        </DataList>
    </>).done({ moduleId: 'doors' });

/// <reference path="../../dts/globals.d.ts" />
/// <reference path="entities.d.ts" />

import { DepartmentsController as actions } from "./api";
import { Field, IOptionsForCRUD, TreeList, ITreeListNode, BaseComponent, Utils } from '@organic-ui';
import { DataTreeList, DataPanel, i18n, DataLookup } from '@organic-ui';
import { AppEntities } from "./entities";
import { departments as departmentListView } from "./department-view";
import { DEPARTMENT_DEFINE_CREATE, DEPARTMENT_DEFINE_DELETE, DEPARTMENT_DEFINE_UPDATE, DEPARTMENT_DEFINE_READ } from './permissions';

const options: IOptionsForCRUD = {
    routeForSingleView: '/view/admin/department/:id',
    routeForListView: '/view/admin/departments',
    pluralName: 'departments', singularName: "department", iconCode: 'fa-sitemap',
    permissionKeys: {
        forCreate: DEPARTMENT_DEFINE_CREATE,
        forDelete: DEPARTMENT_DEFINE_DELETE,
        forUpdate: DEPARTMENT_DEFINE_UPDATE,
        forRead: DEPARTMENT_DEFINE_READ
    },
    filterOptions:{
        liveMode:true
    }
};
export const departments = reinvent('frontend:crud', { actions, options }).
    singleView(({ binding }) =>
        (<>
            <DataPanel header={i18n("primary-fields")} primary className="medium-fields">
                <Field accessor={binding.parentId}>
                    <DataLookup popupMode={DataLookup.Modal} source={departmentListView} />
                </Field>
                <Field accessor={binding.code} required />
                <Field accessor={binding.name} required />
                <Field accessor={binding.description} />
            </DataPanel>
        </>))
    .listView(({ binding }) =>
        (<>
            <DataTreeList >
                <Field accessor={binding.id} role="key" />
                <Field accessor={binding.parentId} role="parentKey" />
                <Field accessor={binding.name} role="text" />
            </DataTreeList>
        </>))
    .done({ moduleId: 'departments' });

/// <reference path="../../dts/globals.d.ts" />
/// <reference path="entities.d.ts" />
import { listView as roleListView } from './role-view'
import { UsersController as actions } from "./api";
import { DataList, DataPanel, DataListPanel, DataLookup, IOptionsForCRUD, Checkbox, TextField } from "@organic-ui";
import { i18n, Field , Switch} from "@organic-ui";

import { crudPermissions, USER_DEFINE_CREATE } from './permissions';
import { ListView as dataItemGroupListView } from './dataitemgroup-view';


const options: IOptionsForCRUD = {
    routeForSingleView: '/view/admin/user/:id',
    routeForListView: '/view/admin/users',
    pluralName: 'users', singularName: 'user', iconCode: 'fa-user-circle',
    permissionKeys: crudPermissions(USER_DEFINE_CREATE)
};
export const users = reinvent('frontend:crud', { actions, options })
    .singleView(({ binding }) => <>
        <DataPanel header={i18n("primary-fields")} primary className="medium-fields" >
            <Field accessor={binding.active} defaultValue={true}  >
                <Switch />
            </Field> <hr />
            <span style={{ flex: '1', minWidth: '100%' }} />



            <Field accessor={binding.username} required  >
                <TextField />
            </Field>
            <Field accessor={binding.fullName} required  >
                <TextField />
            </Field>



            <Field accessor={binding.password}   >
                <TextField type="password" />
            </Field>
            <Field accessor={binding.confirmPassword}   >
                <TextField type="password" />
            </Field>

            
        </DataPanel>
        <DataPanel header="extra" className="half-column-fields">
                
        <Field accessor={binding.rolesIds}     >
                <DataLookup multiple source={roleListView} popupMode={DataLookup.Modal} />
            </Field>

            <Field accessor={binding.dataItemGroupIds} label={i18n("user__dig")} >
                <DataLookup source={dataItemGroupListView} multiple popupMode={DataLookup.Modal} />
            </Field>

        </DataPanel>

    </>)
    .listView(({ binding }) => <>
        <DataList>
            <Field accessor={binding.fullName} />
            <Field accessor={binding.username} />
            <Field accessor={binding.active} />
        </DataList>
    </>)
    .done({ moduleId: 'users' })
