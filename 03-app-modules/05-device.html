# ماژول دستگاه ها در فرانت device-view.tsx


 





























































































































































/// <reference path="../../dts/globals.d.ts" />
/// <reference path="entities.d.ts" />


import { departments as departmentListView } from "./department-view";
import { DevicesController as actions } from "./api";
import { DataItemGroupsController as actionsDIG } from "./api";
import { DeviceEntranceModes, DeviceMatchingModes, DeviceConnectionModes, DeviceModels } from "./zero-data-structures";
import { DataLookup, Field, ComboBox, IOptionsForCRUD, Utils, Switch, Button, AppUtils, FilterPanel } from "@organic-ui";
import { DataList, DataPanel, TextField, Checkbox } from "@organic-ui";
import { crudPermissions, DEVICE_DEFINE_CREATE, DEVICE_DEFINE_DELETE, DEVICE_DEFINE_UPDATE, DEVICE_DEFINE_READ } from './permissions';
import { ListView as dataItemGroupListView } from './dataitemgroup-view';
import { EntityTypesDIG } from './zero-data-structures';
import { ScanDeviceDialog } from "./dialogs/scan-device-dialog";

const { i18n } = OrganicUI;


const options: IOptionsForCRUD =
{
    routeForSingleView: '/view/admin/device/:id',
    routeForListView: '/view/admin/devices',
    pluralName: 'devices', singularName: "device", iconCode: 'fa-calculator',
    permissionKeys: {
        forCreate: DEVICE_DEFINE_CREATE,
        forDelete: DEVICE_DEFINE_DELETE,
        forUpdate: DEVICE_DEFINE_UPDATE,
        forRead: DEVICE_DEFINE_READ
    },
    filterOptions: {

        liveMode: true
    }
};
export const devices = reinvent('frontend:crud', { options, actions })
    .singleView
    (({ binding }) => <>

        <DataPanel header={i18n("primary-fields")} primary className="medium-fields" >
            <Field accessor={binding.active} defaultValue={true}  >
                <Switch />
            </Field> <hr />
            <Field accessor={binding.code}>
                <TextField type="text" />
            </Field>
            <Field accessor={binding.name} required >
                <TextField type="text" />
            </Field>
            <Field accessor={binding.serial} required>
                <TextField type="text" />
            </Field>
            <Field accessor={binding.model} required >
                <ComboBox items={Utils.enumToIdNames(DeviceModels)} />
            </Field>

        </DataPanel>
        <DataPanel header="settings" className="medium-fields"  >
            <Field accessor={binding.useDhcp}    >
                <Switch />
            </Field> <hr />
            <Field accessor={binding.connectionMode}    >
                <ComboBox items={Utils.enumToIdNames(DeviceConnectionModes)} />
            </Field>
            <Field accessor={binding.matchingMode}    >
                <ComboBox items={Utils.enumToIdNames(DeviceMatchingModes)} />
            </Field>
            <Field accessor={binding.port} required  >
                <TextField type="text" />
            </Field>

            <Field accessor={binding.ip}     >
                <TextField />
            </Field>
            <Field accessor={binding.subnet}    >
                <TextField />
            </Field>
            <Field accessor={binding.gateway}>
                <TextField />
            </Field>

        </DataPanel>

        <DataPanel header="other-settings" className="medium-fields">


            <Field accessor={binding.entranceMode}  >
                <ComboBox items={Utils.enumToIdNames(DeviceEntranceModes)} />
            </Field>

            <Field accessor={binding.syncLog}>
                <Checkbox />
            </Field>
            <Field accessor={binding.syncTasks}  >
                <Checkbox />
            </Field>
            <Field accessor={binding.syncTimeByServer}  >
                <Checkbox />
            </Field>

            <Field accessor="syncLogStartDate" />
            <Field accessor="departmentId">
                <DataLookup popupMode={DataLookup.Modal} source={departmentListView} />
            </Field>
            <Field accessor={binding.dataItemGroupsIds} label={i18n("device__dig")} >
                <DataLookup source={dataItemGroupListView} multiple popupMode={DataLookup.Modal}
                    customReadList={actionsDIG.readListModular}
                    filterModelAppend={[{ value: EntityTypesDIG.Device, fieldName: "moduleKey", op: "=" }]}
                />
            </Field>
            <br />
            <br />
            <br />
            <br />
            <br />


        </DataPanel>
    </>)
    .listView(({ binding, reload }) => (<>
        <Button onClick={() => AppUtils.showDialog(<ScanDeviceDialog afterSuccess={reload} />, { noClose: true, })} data-role="nav">{i18n('network-scan')}</Button>
        <FilterPanel liveMode>
            <Field accessor={binding.name} />
            <Field accessor={binding.code} />
            <Field accessor={binding.serial} />
            <Field accessor={binding.ip} />
            <Field accessor={binding.model}>
                <ComboBox items={Utils.enumToIdNames(DeviceModels)} />
            </Field>

            <Field accessor={binding.active} />

            {/* <Field accessor={binding.departmentId} filterData={{ fieldType: "enum" }} operators={['eq']} defaultOperator="eq">
                <DataLookup popupMode={DataLookup.Modal} source={departmentListView} />
            </Field> */}
            {/* <Field accessor={binding.connectionStatusStr} />
            <Field accessor={binding.lastAttendanceStr} /> */}
        </FilterPanel>
        <DataList>
            <Field accessor={binding.name} />
            <Field accessor={binding.code} />
            <Field accessor={binding.serial} />
            <Field accessor={binding.ip} />
            <Field accessor={binding.model} required columnProps={{ minWidth: 150 }}>
                <ComboBox items={Utils.enumToIdNames(DeviceModels)} />
            </Field>

            <Field accessor={binding.active} />

            <Field accessor={binding.departmentId} filterData={{ fieldType: "enum" }} operators={['eq']} defaultOperator="eq">
                <DataLookup popupMode={DataLookup.Modal} source={departmentListView} />
            </Field>
            <Field accessor={binding.connectionStatusStr} avoidSort />
            <Field accessor={binding.lastAttendanceStr} avoidSort />
        </DataList>
    </>)
    )
    .done({ moduleId: 'devices' });

رجوع به تعاریف خودمان – ما چندین فایل مهم تعریف داریم که ساختار پروژه به آن اشاره شده است . فایل تعریف پسوند شان می باشد
لینک با ماژول دپارتمان ها - بخاطر DataLookup
لینک با api مربوط به دستگاه
لینک با ماژول داده های سفارشی
لینک با ماژول هسته سیستم `organic-ui`
تنظیمات مربوط ماژول مشخص میکند - این ماژول از نوع CRUD می باشد - مجوز ها نیز در قسمت permissions مشخص شده است
تعریف یک ماژول `crud` بوسیله `reinvert`
کد های مربوط به نمایش تکی single-view
کد های مربوط به نمایش لیست - موارد مربوط به فیلتر و دکمه ها توسط فریم ورک انجام می شود
